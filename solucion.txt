 eje 4:

SELECT usairports.City,flights.colYear,flights.colMonth,flights.ArrTime FROM flights
LEFT JOIN usairports
ON usairports.IATA = flights.Origin
GROUP BY 
City ASC,
flights.colYear,
flights.colMonth

SELECT flights.UniqueCarrier,flights.colYear,flights.colMonth,flights.Cancelled FROM `flights`
WHERE flights.Cancelled != 0
GROUP BY
flights.UniqueCarrier
ORDER BY
flights.Cancelled DESC;
